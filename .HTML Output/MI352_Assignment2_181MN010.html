def sigma_t(p,q,r,k,beta):
  beta_radian = math.radians(beta)
  csb = math.cos(beta_radian)**2
  cs2b = math.cos(math.radians(2*beta))**2
  ssb = math.sin(beta_radian)**2
  
  l1 = (p**2 + 6*r*q)*ssb
  l2 = (q**2 + 6*r*p)*csb
  l3 = 6*r*(p+q) *cs2b
  l4 = 9*(r**2)
  l = l1 + l2 -l3 + l4

  r1 = (k+1)*((p**2)*(ssb) + (q**2)*(csb) - 9*(r**2))
  r2 = ((p**2 - q**2)*(k+1) - ((p+q)**2)*(k-1))/(p+q-2*r)
  r2m = ((p-3*r)*ssb - (q-3*r)*csb)
  r = r1 - r2*r2m

  sigmat = r/l
  return(sigmat)

p_elps = 2
q_elps = 1
r_elps = 0

sigmat_values = [[0 for j in range(k.size)] for i in range(beta.size)]
for j in range(k.size):
   for i in range(beta.size):
     sigmat_values[i][j] = sigma_t(p_elps,q_elps,r_elps,k[j],beta[i])

beta_transformed_x = np.zeros(beta.size)

for i in range(beta.size):
  beta_radian = math.radians(beta[i])
  beta3_radian = math.radians(3*beta[i])
  beta_transformed_x[i] = p_elps*(math.cos(beta_radian)) + r_elps*(math.cos(beta3_radian))

plt.figure(figsize=(10,12))
for i in range(k.size):
  sigmatt_values =  [(row[i-1]) for row in sigmat_values]
  plt.plot(beta_transformed_x,sigmatt_values, label = round(k[i],3))

plt.xlabel('theta ----->')
plt.ylabel('sigma_t/S_y')
plt.legend(title = 'k-values')
plt.title('sigma_t/S_y vs Theta for Ellipse (2:1)', fontsize = 15)
plt.show()
